{#असश्चतः । शतधारा अभिश्रियो हरिं नवन्तेऽव ता उदन्युवः#}
({%Sāyaṇa:%} {#अभिश्रयः । अभितः सोमं श्रयन्त्यः#}); or {%Ṛgv.,%}

{#रसा दधीत वृषभम्#} ({%Sāyaṇa:%} {#अभिश्रियमभिश्रयन्तम्;#} but

venerated everywhere; e. g. {%Ṛgv., Sāmav., Vājas.:%} {#घृतवती#}
{#भुवनानामभिश्रियोर्वी पृथ्वी मधुदुधे सुपेशसा#} ({%Sāyaṇa%} and
{%Mahīdh.:%} {#अभिश्रिया = अभिश्रयणीये#} or {#आश्रयणीये#}); or {%Ṛgv.,%}
{%Vājas.:%} {#वैश्वानरस्य सुमतौ स्याम राजा हि कं भुवनानामभि-#}
{#श्रीः#} ({%Sāyaṇa:%} {#अभिश्रीरभिश्रयणीय आभिमुख्येन सेवितव्यः;#}
